1. Un abrevadero de longitud tiene una seccin transversal en forma de semicrculo con radio . (Consulte la figura adjunta.) Cuando se llena con agua hasta una distancia a partir de la parte superior, el volumen de agua es $$ V = L[0.5\pi r^{2}-r^{2}arcsen (h/r)-h(r^{2}-h^{2})^{1/2}]$$

Suponga que = 10 , = 1 y = 12.4 $^3$. Encuentre la profundidad del agua en el abrevadero dentro de 0.01c.

In[3]:
import math

# Datos 
L = 10      # cm
r = 1       # cm
V_deseado = 12.4  # cm^3
tolerancia = 0.01  # cm

# Funcin del volumen V(h) - V_deseado
def f(h):
    if h < 0 or h > r:
        return float('inf')  # fuera de dominio
    termino1 = 0.5 * math.pi * r**2
    termino2 = r**2 * math.asin(h / r)
    termino3 = 0.5 * h * math.sqrt(r**2 - h**2)
    V = L * (termino1 - termino2 - termino3)
    return V - V_deseado

# Mtodo de biseccin
def biseccion(f, a, b, tol=1e-5, max_iter=100):
    fa = f(a)
    fb = f(b)
    if fa * fb > 0:
        print("No hay cambio de signo, no se puede aplicar biseccin.")
        return None

    for i in range(max_iter):
        c = (a + b) / 2
        fc = f(c)

        if abs(fc) < tol or (b - a) / 2 < tol:
            return c

        if fa * fc < 0:
            b = c
            fb = fc
        else:
            a = c
            fa = fc

    return (a + b) / 2  # retorno final si no se alcanza la tolerancia

# Ejecucin de la biseccin en el intervalo [0, 1]
h_aproximado = biseccion(f, 0, r, tol=0.01)

print(f"La profundidad del agua h  {h_aproximado:.4f} cm")
